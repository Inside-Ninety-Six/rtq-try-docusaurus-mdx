                                                        &= 72 \\\\
\dfrac{3} {4} \times 72                                 &= \dfrac{3 \times 72} {4} \\\\
                                                        &= \dfrac{3 \times 18 \times 4} {4} \\\\
                                                        &= \dfrac{3 \times 18 \times \cancel{4}} {\cancel{4}} \\\\
                                                        &= 54                                                        
\end{aligned}
