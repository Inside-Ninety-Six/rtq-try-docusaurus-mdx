a                            &= \dfrac {360} {10 \times 3.6} \\\\
a                            &= \dfrac {360} {36} \\\\
a                            &= \dfrac {36 \times 10} {36} \\\\
a                            &= \dfrac {\cancel {36} \times 10} {\cancel {36}} \\\\
a                            &= 10
\end{aligned}
