&= \dfrac {47} {60} \\\\

\end{aligned}

&= \dfrac{36} {100} + \dfrac{64} {100} \\\\
&= \dfrac{36 + 64} {100} \\\\
&= \dfrac{100} {100} \\\\
&= \dfrac{\cancel {100}} {\cancel {100}} \\\\
&= 1
\end{aligned}
