$
\begin{aligned}
a \times 15 - 7                     &= 38 \\\\
a                                   &= \dfrac {38 + 7} {15}  \\\\
a                                   &= \dfrac {15 \times 3} {15} \\\\
a                                   &= \dfrac {\cancel {15} \times 3} {\cancel {15}} \\\\
a                                   &= 3
\end{aligned}
$

</div>
</div>
<div class='answers'>
<div class='answer'>

$3$
