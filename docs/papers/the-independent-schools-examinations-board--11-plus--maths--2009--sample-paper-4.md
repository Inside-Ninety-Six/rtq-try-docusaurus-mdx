\begin{aligned}

&= \dfrac {60} {3} \\\\\
&= \dfrac {20 \times 3} {3} \\\\\
&= \dfrac {20 \times \cancel 3} {\cancel 3} \\\\\
&= 20 \  \text {mins}
\end{aligned}
