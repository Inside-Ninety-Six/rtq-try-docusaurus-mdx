&= \dfrac {2} {5} \times 30 \\\\
&= \dfrac {2} {5} \times {5 \times 6} \\\\
&= \dfrac {2 \times 5 \times 6} {5} \\\\
&= \dfrac {2 \times \cancel {5} \times 6} {\cancel {5}} \\\\
&= 2 \times 6 \\\\
&= 12
\end {aligned}
$
