<div class='working'>

\begin{aligned}

&= \dfrac {3} {8} \times \dfrac {4} {1} \\\\
&= \dfrac {3 \times 4} {8 \times 1} \\\\
&= \dfrac {3 \times 4} {2 \times 4} \\\\
&= \dfrac {3 \times \cancel{4}} {2 \times \cancel{4}} \\\\
&= \dfrac {3} {2}
\end{aligned}
$
